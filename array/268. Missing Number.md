class Solution {
    public int missingNumber(int[] nums) {
        Arrays.sort(nums);
        int res = 0;
        //special case 1
        if(nums.length == 1){
            if(nums[0] == 1){
                return 0;
            }
            else{
                return 1;
            }
        }
        //special case 2
        if(nums[nums.length - 1] == nums.length -1){
            return nums.length;
        }
        for(int i = 0;  i <= nums.length - 2;i++){
            if(nums[i] + 1 != nums[i+1]){
                res = nums[i] + 1;
                break;
            }
        }
        return res;
    }
}
